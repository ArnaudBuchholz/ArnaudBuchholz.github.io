    Subclassing an ES6 class

    const gpf = require("gpf-js");
class A {
  constructor () {
    this._a = "A";
  }
}
const B = gpf.define({
	$class: "B",
	$extend: A,
	constructor: function () {
	  this.$super();
	  this._b = "B";
	}
});

    JavaScript classes

    The 'old' way

    There are many patterns, this is the one used in GPF-JS
    Declaration
    function OldA () {
  this._a = "A";
}

OldA.prototype.a = function () {
  return this._a;
};
    Subclassing
    function OldB () {
  OldA.apply(this, arguments);
  this._b = "B";
}

OldB.prototype =
	Object.create(OldA.prototype);

OldB.prototype.b = function () {
  return this._b;
};

    The ES6 way

    Declaration
    class Es6A {
	constructor () {
    this._a = "A";
  }

  a () {
    return this._a;
  }
};
    Subclassing
    class Es6B extends Es6A {
  constructor () {
    super();
    this._b = "B";
  }

  b () {
    return this._b;
  }
}

    Subclassing an Old class with ES6

    class ExtendOldWithEs6B extends OldA {
    constructor () {
      super();
      this._b = "B";
    }

    b () {
      return this._b;
    }
}
